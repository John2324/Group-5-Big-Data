scp /Users/mario/Desktop/iowa/pt2.csv mmonter3@54.187.148.25:/home/mmonter3

hdfs dfs -mkdir alcohol

-put the csv file into the alcohol folder- 
hdfs dfs -put pt2.csv alchol
