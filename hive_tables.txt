CREATE VIEW dollarvolume
AS SELECT item_name, bottles_sold, sale_dollar, sale_volume_gallon
FROM iowa
UNION ALL
SELECT item_name, bottles_sold, sale_dollar, sale_volume_gallon
FROM iowa2;
